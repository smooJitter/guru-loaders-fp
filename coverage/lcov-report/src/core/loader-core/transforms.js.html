/**
 * Named transforms for loader pipelines (sync and async).
 *
 * - Co-locate sync and async transforms for clarity and maintainability.
 * - Use clear naming (e.g., asyncEnrichFromAPI) for async transforms.
 * - Both types can be used in composeTransforms and transformRegistry.
 *
 * | Transform Type      | Example Purpose                              |
 * |---------------------|----------------------------------------------|
 * | normalizeMod        | Consistent array/object/factory output       |
 * | injectContext       | Add context/services to each module          |
 * | addMetadata         | Add type, timestamp, etc.                    |
 * | legacyAdapter       | Adapt old modules to new format              |
 * | wrapWithGuard       | Add logging/authorization to handlers        |
 * | validationFilter    | Remove invalid modules early                 |
 * | asyncEnrichFromAPI  | (Async) Enrich events with remote data       |
 */
 
import { isNil } from 'ramda';
 
/**
 * Utility: Apply a function to a single mod or an array of mods.
 * @param {any|any[]} mods - A single mod or array of mods.
 * @param {Function} fn - Function to apply.
 * @returns {any|any[]}
 */
const mapMods = (mods, fn) =>
  Array.isArray(mods) ? mods.map(fn) : fn(mods);
 
/**
 * DRY helper: create both single and array versions of a transform.
 */
const makeTransform = (singleFn) => ({
  single: singleFn,
  array: (mods, ctx) => mapMods(mods, m => singleFn(m, ctx)),
  main: (mods, ctx) => mapMods(mods, m => singleFn(m, ctx)), // default export
});
 
/**
 * Normalize a module to a consistent shape for loader-core pipelines.
 *
 * - If the module has a `default` export, use that (ESM/CJS compatibility).
 * - If the export is a function, call it with the loader context (for context-aware factories).
 * - If the result is an array, return it as-is (supports withNamespace, multi-action, or multi-handler patterns).
 * - If the result is a single object, wrap it in an array (ensures downstream code always receives an array).
 *
 * This enables loader-core to handle:
 *   - Factory exports: `export default (ctx) => withNamespace('foo', { ... })`
 *   - Array exports:   `export default [ ... ]`
 *   - Single object:   `export default { ... }`
 *
 * The result is always an array of registry objects (e.g., `{ namespace, name, method }`).
 *
 * @param {any} mod - The imported module (may be ESM/CJS, factory, array, or object)
 * @param {object} ctx - The loader context
 * @returns {Array} Array of registry objects for the loader pipeline
 */
const normalizeMod = (mod, ctx) => {
  // Unwrap default export if present (ESM/CJS compatibility)
  const modVal = !isNil(mod) && mod.default !== undefined ? mod.default : mod;
  // If it's a function, call it with context (context-aware factory)
  const events = typeof modVal === 'function' ? modVal(ctx) : modVal;
  // Always return an array (wrap single object if needed)
  return Array.isArray(events) ? events : [events];
};
const normalizeModSingle_ = (mod, ctx) => {
  const modVal = !isNil(mod) && mod.default !== undefined ? mod.default : mod;
  return typeof modVal === 'function' ? modVal(ctx) : modVal;
};
export const { single: normalizeModSingle, array: normalizeModArray } = makeTransform(normalizeModSingle_);
export { normalizeMod };
 
/**
 * Inject context.services into mod(s).
 */
const injectContextSingle_ = (mod, ctx) =>
  isNil(mod) || typeof mod !== 'object'
    ? mod
    : { ...mod, services: ctx?.services };
export const { single: injectContextSingle, array: injectContextArray, main: injectContext } = makeTransform(injectContextSingle_);
 
/**
 * Add metadata to mod(s).
 */
const addMetadataSingle_ = (mod) =>
  isNil(mod) || typeof mod !== 'object'
    ? mod
    : { ...mod, type: 'event', timestamp: Date.now() };
export const { single: addMetadataSingle, array: addMetadataArray, main: addMetadata } = makeTransform(addMetadataSingle_);
 
/**
 * Adapt legacy mods with a methods property.
 */
const legacyAdapterSingle_ = (mod) =>
  mod && mod.methods
    ? { ...mod, event: mod.methods.main || mod.event }
    : mod;
export const { single: legacyAdapterSingle, array: legacyAdapterArray, main: legacyAdapter } = makeTransform(legacyAdapterSingle_);
 
/**
 * Wrap mod.event with a guard/logging function.
 */
const wrapWithGuardSingle_ = (mod) =>
  isNil(mod) || typeof mod !== 'object' || typeof mod.event !== 'function'
    ? mod
    : {
        ...mod,
        event: (...args) => {
          console.log(`[GUARD] Executing event: ${mod.name}`);
          return mod.event(...args);
        }
      };
export const { single: wrapWithGuardSingle, array: wrapWithGuardArray, main: wrapWithGuard } = makeTransform(wrapWithGuardSingle_);
 
/**
 * Filter out invalid mods (must have name:string and event:function).
 */
const validationFilterSingle = (mod) =>
  mod && typeof mod.name === 'string' && typeof mod.event === 'function' ? mod : null;
const validationFilterArray = (mods) =>
  (Array.isArray(mods) ? mods : [mods]).filter(validationFilterSingle).filter(Boolean);
export const validationFilter = validationFilterArray;
export { validationFilterSingle, validationFilterArray };
 
/**
 * Asynchronously enrich mod(s) with extra data.
 */
const asyncEnrichFromAPISingle = async (mod, ctx) => {
  if (!isNil(mod) && typeof mod === 'object') {
    const extra = await Promise.resolve({ info: `Extra for ${mod.name}` });
    return { ...mod, extra };
  }
  return mod;
};
const asyncEnrichFromAPIArray = async (mods, ctx) =>
  Array.isArray(mods)
    ? Promise.all(mods.map(m => asyncEnrichFromAPISingle(m, ctx)))
    : asyncEnrichFromAPISingle(mods, ctx);
export const asyncEnrichFromAPI = asyncEnrichFromAPIArray;
export { asyncEnrichFromAPISingle, asyncEnrichFromAPIArray };
 
/**
 * Registry of named transforms (sync and async).
 */
export const transformRegistry = {
  normalizeMod,
  injectContext,
  addMetadata,
  legacyAdapter,
  wrapWithGuard,
  validationFilter,
  asyncEnrichFromAPI,
  // Single-object
  normalizeModSingle,
  injectContextSingle,
  addMetadataSingle,
  legacyAdapterSingle,
  wrapWithGuardSingle,
  validationFilterSingle,
  asyncEnrichFromAPISingle,
  // Array-based explicit
  normalizeModArray,
  injectContextArray,
  addMetadataArray,
  legacyAdapterArray,
  wrapWithGuardArray,
  validationFilterArray,
  asyncEnrichFromAPIArray,
};
 
/**
 * Compose transforms by name (from the registry) and/or by function.
 */
export const composeTransforms = (transforms = []) => async (mod, ctx) => {
  let acc = mod;
  for (const t of transforms) {
    if (typeof t === 'string' && transformRegistry[t]) {
      acc = await transformRegistry[t](acc, ctx);
    } else if (typeof t === 'function') {
      acc = await t(acc, ctx);
    }
  }
  return acc;
};
 
/**
 * Utility to select correct transform pipeline based on registry type.
 */
export const selectTransformsForRegistry = (transforms, registryType) => {
  if (['event', 'pipeline', 'tagged', 'flat', 'composite'].includes(registryType)) {
    return transforms;
  } else {
    return transforms.map(fn => (mod, ctx) => {
      if (Array.isArray(mod)) {
        return fn(mod[0], ctx);
      }
      return fn(mod, ctx);
    });
  }
};
